<!-- Use these tags to write comments. 
    The index.html is typically the homepage of any web
application.  -->
<!DOCTYPE html>
<!-- This is the DOCTYPE declaration. It tells the browsers
what version of html they are interpreting. If you just use html,
it will default to html 5. -->
<html>
<!-- The root tag for any html file will be <html> -->
    <head>
        <!-- The <head> tag is used to store metadata about the page
            such as, page title, or charset encodings. -->
        <meta charset="UTF-8">
        <title>My Fantastic Website :D</title>
        <!-- The title tag is simply that. It will display on the top of pages, and tabs. -->
    </head>
    <body>
        <!-- The <body> tag is what is actually displayed to the user
            on the browser. -->
        <h1>WELCOME TO MY WEBSITE</h1>
        <!-- Header1 tag -->
        <p>Please stay a while, and listen.</p>
        <!-- paragraph tag -->
    </body>
</html
